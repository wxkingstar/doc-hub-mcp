## 旧版jweixin调用
### 示例代码
```javascript
wx.onLocationChange( 
 function(res) {
 if(res.errMsg == "auto:location:report:ok"){
 var latitude = res.latitude; // 纬度，浮点数，范围为90 ~ -90
 var longitude = res.longitude; // 经度，浮点数，范围为180 ~ -180。
 var speed = res.speed; // 速度，以米/每秒计
 var accuracy = res.accuracy; // 位置精度
 var lbsIndex = 0;
 }else {
 //错误处理
 }
 }
);
```
