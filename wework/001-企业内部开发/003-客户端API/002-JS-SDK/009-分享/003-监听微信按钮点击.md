## 旧版jweixin调用
### 示例代码
```javascript
wx.onMenuShareWechat({
 title: '', // 分享标题
 desc: '', // 分享描述
 link: '', // 分享链接
 imgUrl: '', // 分享图标
 enableIdTrans: 1, // 是否开启id转译，不填默认为0
 success: function () {
 // 用户确认分享后执行的回调函数
 },
 error: function(res) {
 if(res.errMsg.indexOf('no permission') > 0){
 alert('未agentConfig')
 }
 },
 cancel: function () {
 // 用户取消分享后执行的回调函数
 }
});
```
