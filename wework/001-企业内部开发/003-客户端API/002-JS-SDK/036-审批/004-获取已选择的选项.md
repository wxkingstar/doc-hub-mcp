```
wx.invoke('getApprovalSelectedItems', {
 "key": key, // 字符串，从 URL 中获取到的 key
}, (res)=> {
 if (res.err_msg === "getApprovalSelectedItems:ok") {
      try {
        // !!!注意：旧版中res.selectedData 为 JSON 字符串。
        const selectedData = JSON.parse(res.selectedData)
      } catch() {
        // 注意可能为空
      }
 }
});
```
