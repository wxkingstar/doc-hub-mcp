## 旧版jweixin调用

### 代码示例
```
wx.invoke('getCurCorpGroupContact', {}, function(res) {
 if (res.err_msg == 'getCurCorpGroupContact:ok') {
 corpId = res.corpId // 返回当前联系人的企业corpId
 userId = res.userId // 返回当前联系人userId
 } else {
 // 错误处理
 }
})
```
